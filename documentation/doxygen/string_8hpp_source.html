104  //\brief Map encoding to string
105  //\param[in] e encoding
106  //\param[in] c code page in case if and ISO 8859 encoding
107  //\return the name of the encoding
108  static const char* encodingName( Encoding e, unsigned char c=0)
109  {
110  static const char* ar[] = {"ISO8859","UTF-8","UTF-16BE","UTF-16LE","UCS-2BE","UCS-2LE","UCS-4BE","UCS-4LE"};
111  static const char* cp[] = {"ISO8859-1","ISO8859-2","ISO8859-3","ISO8859-4","ISO8859-5","ISO8859-6","ISO8859-7","ISO8859-8","ISO8859-9"};
112  if (e == ISO8859 && c)
113  {
114  if (c > 9) return 0;
115  return cp[ c];
116  }
117  else
118  {
119  return ar[e];
120  }
121  }
122  //\brief Get the encoding of this string as string
123  //\return the name of the encoding
