347  static void createMetatable( lua_State* ls)
348  {

358  {

360  if (rt == 0) throw std::bad_alloc();

366  //\brief does nothing because the LUA garbage collector does the job.

368  void operator delete (void *, lua_State*) {}

370  static int destroy( lua_State* ls)
371  {
372  LuaArithmeticType *THIS = (LuaArithmeticType*)lua_touserdata( ls, 1);
373  if (THIS) THIS->~LuaArithmeticType();

377  static void push_luastack( lua_State* ls, const ArithmeticType& o)
378  {
379  try
380  {
381  (void*)new (ls) LuaArithmeticType( o);
382  }
383  catch (const std::exception&)
384  {
385  luaL_error( ls, "memory allocation error in lua context");
386  }
387  }

389  static ArithmeticType* get( lua_State* ls, int index)
390  {
391  ArithmeticType* rt = (ArithmeticType*) luaL_testudata( ls, index, MetaTypeInfo::metatableName());
392  if (!rt)
393  {
394  const char* arg = lua_tostring( ls, index);
395  if (arg == 0) throw std::runtime_error( "cannot convert argument to arithmetic type");
396  LuaExceptionHandlerScope escope(ls);
397  {
398  (void)new (ls) LuaArithmeticType( std::string(arg));
399  }
400  rt = (ArithmeticType*) luaL_checkudata( ls, -1, MetaTypeInfo::metatableName());
401  }
402  return rt;
