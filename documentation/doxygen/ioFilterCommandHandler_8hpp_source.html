77  {
78  if (m_inputfilter->state() != langbind::InputFilter::Start)
79  {
80  throw std::runtime_error( "cannot change input filter after first read");
81  }
82  // assign the rest of the input to the new filter attached:

85  bool chunkend;
86  m_inputfilter->getRest( chunk, chunksize, chunkend);
87  langbind::InputFilter* incopy = in->copy();
88  try
89  {

91  }
92  catch (const std::bad_alloc& e)

94  delete incopy;
95  throw e;
96  }
97  catch (const std::runtime_error& e)

102  m_inputfilter.reset( incopy);
103  // synchronize inherited meta data:

107  }
