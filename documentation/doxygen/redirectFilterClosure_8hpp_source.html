88  bool call()
89  {
90  if (!m_inputfilter.get()) throw std::runtime_error( "no input defined for redirecting filter");
91  if (!m_outputfilter.get()) throw std::runtime_error( "no output defined for redirecting filter");
92 
93  for (;;) switch (m_state)
94  {
95  case 0:
96  if (!m_inputfilter->getNext( m_elemtype, m_elem))
97  {
98  switch (m_inputfilter->state())
99  {
100  case InputFilter::Start:
101  case InputFilter::Open:
102  m_state = (m_do_print_final_close)?2:3;
103  continue;
104 
105  case InputFilter::EndOfMessage:
106  return false;
107 
108  case InputFilter::Error:
109  throw std::runtime_error( m_inputfilter->getError());
110  }
111  }
112  m_state = 1;
113  if (m_elemtype == InputFilter::OpenTag)
114  {
115  ++m_taglevel;
116  }
117  else if (m_elemtype == InputFilter::CloseTag)
118  {
119  --m_taglevel;
120  if (m_taglevel < 0)
121  {
122  m_state = (m_do_print_final_close)?2:3;
123  continue;
124  }
125  }
126  /*no break here!*/
127  case 1:
128  if (!m_outputfilter->print( m_elemtype, m_elem))
129  {
130  switch (m_outputfilter->state())
131  {
132  case OutputFilter::Start:
133  case OutputFilter::Open:
134  throw std::runtime_error( "unknown error in output filter");
135 
136  case OutputFilter::EndOfBuffer:
137  return false;
138 
139  case OutputFilter::Error:
140  throw std::runtime_error( m_outputfilter->getError());
141  }
142  }
143  m_state = 0;
144  continue;
145  case 2:
146  if (!m_outputfilter->print( m_elemtype, m_elem))
147  {
148  switch (m_outputfilter->state())
149  {
150  case OutputFilter::Start:
151  case OutputFilter::Open:
152  throw std::runtime_error( "unknown error in output filter");
153 
154  case OutputFilter::EndOfBuffer:
155  return false;

158  throw std::runtime_error( m_outputfilter->getError());
159  }
160  }
161  m_state = 3;
162  /*no break here!*/
163  default:
