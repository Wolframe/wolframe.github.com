141 class InputBlock :public MemBlock
142 {
143 public:
148  // See method getEoD(iterator)
149  //
150  // The end of data is marked with a sequence "\n.\r" or "\n.\n".
151  // If such a sequence appears by accident in the content it has to be escaped with a dot as "\n..\r" or "\n..\r".
152  // The additional dot after the end of line escapes any other character sequence too.
153  // As consequence every line starting with a dot has to be escaped.
155  struct EoD
156  {
159  enum State
160  {
161  SRC, //< parsing content
162  CR, //< detected an CarriageReturn in state SRC, belonging either to a CRLF or a LF sequence
163  CR_LF, //< detected an LineFeed in state SRC or CR, belonging either to a CRLF or a LF sequence
164  CR_LF_DOT, //< detected a '.' (dot) in state CR_LF
165  CR_LF_DOT_CR, //< detected a CarriageReturn CR in state CR_LF_DOT. Here we assume that end of data is reached
166  CR_LF_DOT_CR_LF //< detected a LineFeed LF in state CR_LF_DOT or CR_LF_DOT_CR. Here we assume at last that end of data is reached
167  };
168  };
