139 // The buffer implements the STL back insertion sequence interface required by the protocol parser.
140 //
141 // This buffer splits the input, A sequence of characters by blanks and it parses escaping and quoted strings in the following way:
142 // Escaping is done with backslash, strings can be single or double quoted.
143 // The maximum number of arguments parsed is fix (16). More arguments are appended to the 16th argument.
144 // A string is starting with a quote and terminating with the same non escaped quote or with the end of line.
145 // Only blanks are splitting elements.
146 //
148 template <class Buffer>

195  void push_back( char ch)
196  {
197  switch (m_state)
198  {
199  case EndToken:
200  switch (ch)
201  {
202  case '\'': m_state = SQContent; break;
203  case '\"': m_state = DQContent; break;
204  case '\\': m_state = ContentEsc; break;
205  case ' ': m_content->push_back(0); m_state = Empty; break;
206  default: m_state = Content; m_content->push_back(ch); break;
207  }
208  break;
209 
210  case Empty:
211  switch (ch)
212  {
213  case '\'': m_state = SQContent; openArg(); break;
214  case '\"': m_state = DQContent; openArg(); break;
215  case '\\': m_state = ContentEsc; openArg(); break;
216  case ' ': break;
217  default: m_state = Content; openArg(); m_content->push_back(ch); break;
218  }
219  break;
220 
221  case Content:
222  switch (ch)
223  {
224  case '\'': m_state = SQContent; break;
225  case '\"': m_state = DQContent; break;
226  case '\\': m_state = ContentEsc; break;
227  case ' ': m_content->push_back(0); m_state = Empty; break;
228  default: m_content->push_back(ch); break;
229  }
230  break;
231 
232  case ContentEsc:
233  m_state = Content;
234  m_content->push_back(ch);
235  break;
236 
237  case SQContent:
238  switch (ch)
239  {
240  case '\'': m_state = EndToken; break;
241  case '\\': m_state = SQContentEsc; break;
242  default: m_content->push_back(ch); break;
243  }
244  break;

256  default: m_content->push_back(ch); break;
257  }
