122 template <typename TYPE>
123 static bool fetchObject_( const traits::vector_&, Context& ctx, FiltermapSerializeStateStack& stk)
124 {
125  const std::vector<typename TYPE::value_type>* obj = (const TYPE*)stk.back().value();
126  std::size_t idx = stk.back().state();
127  if (idx >= obj->size())
128  {
129  stk.pop_back();
130  return false;
131  }
132  const void* ve = &(*obj)[ idx];
