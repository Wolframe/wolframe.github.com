42 //\class SyncCounter
43 //\brief Atomic counter with upper value limit for a synchronized counting of objects
44 class SyncCounter
45  :protected boost::atomic<unsigned int>
46 {
47 public:
48  typedef unsigned int CounterType;
49  typedef boost::atomic<CounterType> Parent;
50 
51  //\brief Constructor
52  explicit SyncCounter( CounterType limitValue_=0, CounterType initialValue_=0)
53  :boost::atomic<CounterType>(initialValue_)
54  ,m_limitValue(limitValue_)
55  {}
56 
57  //\brief Aquire the counter
58  //\return 0 in case of error, the counter instance auqired (>0) in case of success
59  CounterType aquire()
60  {
61  if (m_limitValue)
62  {
63  for (;;)
64  {
65  CounterType val = Parent::load( boost::memory_order_relaxed);
66  if (val >= m_limitValue) return 0;
67  if (Parent::compare_exchange_strong( val, val+1, boost::memory_order_acquire))
68  {
69  return val+1;
70  }
71  }
72  }
73  else
74  {
75  return Parent::fetch_add( 1, boost::memory_order_acquire)+1;
76  }
77  }
78 
79  //\brief Release the counter
80  void release()
81  {
82  Parent::fetch_sub( 1, boost::memory_order_release);
83  }
84

87  // SyncCounter globalCnt;
88  // ....
89  // SyncCounter::ScopedAquire cntscope( globalCnt);
90  // if (cntscope.entered())
91  // {
92  // ... do something that might throw
93  // cntscope.done();

95  //
96  class ScopedAquire
97  {
98  ScopedAquire( SyncCounter& cc_)
99  :m_cc(&cc_)
100  {
101  if (!m_cc->aquire()) m_cc = 0;
102  }
103 
104  ~ScopedAquire()
105  {
106  if (m_cc) m_cc->release();
107  }
108 
109  //\brief Evaluate if the counter could be aquired in the given limits

112  {
113  return m_cc;
114  }
115 
116  //\brief Complete aquiring of the counter: All operations belonging to the counter aquiring that might throw have been completed with success.
117  void done()
118  {
119  m_cc = 0;
120  }
121

124  };
