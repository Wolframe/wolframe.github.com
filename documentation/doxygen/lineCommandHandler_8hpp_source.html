73  int runCommand( std::size_t stateidx, std::size_t cmdidx, void* obj, int argc, const char** argv, std::ostream& out) const
74  {
75  const State& st = get( stateidx);
76  if (cmdidx >= st.m_cmds.size()) throw std::logic_error( "illegal command reference");
77  return st.m_cmds.at(cmdidx)( obj, argc, argv, out);
78  }

80  void defineState( std::size_t se)
81  {
82  if (se != m_statear.size()) throw std::logic_error( "state index defined not matching to enum given as state index");
83  State st;
84  m_statear.push_back( st);
85  }
86 
87  void defineCommand( const char* name_, RunCommand run_)
88  {
89  m_statear.back().defineCommand( name_, run_);
90  }
91 
92  const State& get( std::size_t idx) const
93  {
94  if (idx >= m_statear.size()) throw std::logic_error( "illegal state");
95  return m_statear.at( idx);
96  }
