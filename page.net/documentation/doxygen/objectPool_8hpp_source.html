59 template < typename objectType >
60 class ObjectPool {
61 public:
62  ObjectPool( const unsigned to ) { m_timeout = to; }
63  ObjectPool() { m_timeout = 0; }
64  ~ObjectPool() {
65  boost::lock_guard<boost::mutex> lock( m_mutex );
66  if ( !m_availList.empty() )
67  throw std::runtime_error( "ObjectPool not empty at destruction" );
68  }
69 
70  std::size_t available() const { return m_availList.size(); }
71 
72  objectType get() {
73  while( true ) {
74  boost::unique_lock<boost::mutex> lock( m_mutex );
75  if ( !m_availList.empty()) {
76  objectType obj = m_availList.back();
77  m_availList.pop_back();
78  lock.unlock();
79  return obj;
80  }
81  else {
82  if ( m_timeout == 0 ) {
83  while( m_availList.empty() )
84  m_cond.wait( lock );
85  }
86  else {
87  boost::system_time absTime = boost::get_system_time()
88  + boost::posix_time::seconds( m_timeout );
89  while( m_availList.empty() )
90  if ( ! m_cond.timed_wait( lock, absTime ))
91  throw ObjectPoolTimeout();
92  }
93  }
94  }
95  throw std::logic_error( "Logic error #1 in ObjectPool" );
96  }
97 
98  void add ( objectType obj ) {
99  boost::lock_guard<boost::mutex> lock( m_mutex );
100  m_availList.push_back( obj );
101  m_cond.notify_one();
102  }
103 
104  unsigned timeout() const { return m_timeout; }
105  void timeout( unsigned to ) { m_timeout = to; }
106 private:
107  std::vector< objectType > m_availList;
108  boost::mutex m_mutex;
