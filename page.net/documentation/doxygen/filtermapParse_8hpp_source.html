110 template <typename TYPE>
111 bool parseObject_( const traits::vector_&, langbind::TypedInputFilter&, Context&, FiltermapParseStateStack& stk)
112 {
113  if (stk.back().state())
114  {
115  stk.pop_back();
116  return true;
117  }
118  typedef typename TYPE::value_type Element;
119  boost::value_initialized<Element> val;
120  ((TYPE*)stk.back().value())->push_back( val);
121  stk.back().state( 1);
122  Element* ee = &((TYPE*)(char*)stk.back().value())->back();
123  if (FiltermapIntrusiveProperty<Element>::type() == StructDescriptionBase::Atomic)
124  {
125  stk.push_back( FiltermapParseState( 0, &parseAtomicElementEndTag, 0));
126  }
127  stk.push_back( FiltermapParseState( 0, &FiltermapIntrusiveParser<Element>::parse, ee));
128  return true;
129 }
